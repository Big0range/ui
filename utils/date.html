一些 dayjs 无法实现的特殊需求，如有误请在 GitHub 中提交 issues

import { date } from "@big0range/utils";
// 当前日期的月份最后一天
const last = date.lastDayOfMonth(new Date("2022-01-22"));
// last = 2022-01-31

// 当前日期的月份第一天
const last = date.firstDayOfMonth(new Date("2022-01-22"));
// last = 2022-01-01
